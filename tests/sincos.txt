(for (set i 0) (lt i 1000) (set i (add i 1)) (
    (set alpha (call randint 100))
    (call println "alpha: " alpha " rad")
    (set sin2 (call sin alpha))
    (set sin2 (mult sin2 sin2))
    (call println "sin(alpha)^2 = " sin2)
    (set cos2 (call cos alpha))
    (set cos2 (mult cos2 cos2))
    (call println "cos(alpha)^2 = " cos2)
    (set sum (add sin2 cos2))
    (call println "sin(alpha)^2 + cos(alpha)^2 = " sum)
    (call assert (le (call abs (sub 1 sum)) 0.0000001) "sincos: not 1") 
))
(call println "sincos done")
